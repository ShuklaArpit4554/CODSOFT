#include <iostream>
#include <cstdlib>
#include <ctime>

using namespace std;

int main() {
    srand(time(0));
    int number = rand() % 100 + 1;
    int guess;

    cout << "Guess a number between 1 and 100:\n";

    while (true) {
        cout << "Your guess: ";
        cin >> guess;

        if (guess < number) {
            cout << "Too low.\n";
        } else if (guess > number) {
            cout << "Too high.\n";
        } else {
            cout << "You got it!\n";
            break;
        }
    }

    return 0;
}
